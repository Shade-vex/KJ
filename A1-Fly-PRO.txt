-- Tải script bay
loadstring(game:HttpGet("https://raw.githubusercontent.com/Shade-vex/f/refs/heads/main/Fly.lua.txt", true))()

-- Đợi để đảm bảo PlayerGui đã load
task.wait(1)

-- Tạo GUI
local player = game.Players.LocalPlayer
local gui = Instance.new("ScreenGui", player:WaitForChild("PlayerGui"))
gui.Name = "GhostImageGui"
gui.ResetOnSpawn = false
gui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling

-- Tạo ảnh có thể click
local image = Instance.new("ImageButton")
image.Parent = gui
image.Size = UDim2.new(0, 50, 0, 50)
image.Position = UDim2.new(0.5, -25, 0.5, -25)
image.BackgroundTransparency = 1
image.Image = "rbxassetid://95418274223386" -- ID hợp lệ bạn đã gửi ban đầu

-- Copy link Discord khi click ảnh
image.MouseButton1Click:Connect(function()
    setclipboard("https://discord.gg/gamehayvl")
end)

-- Kéo và thả ảnh
local dragging, dragInput, dragStart, startPos

local function update(input)
    local delta = input.Position - dragStart
    image.Position = UDim2.new(
        startPos.X.Scale, startPos.X.Offset + delta.X,
        startPos.Y.Scale, startPos.Y.Offset + delta.Y
    )
end

image.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = image.Position

        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

image.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
        dragInput = input
    end
end)

game:GetService("UserInputService").InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        update(input)
    end
end)
